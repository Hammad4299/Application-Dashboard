sudo apt-get install supervisor
sudo nano /etc/supervisor/conf.d/posthider-queue-workers.conf

//Content of above file
[program:posthider-queue-workers]
process_name=%(program_name)s_%(process_num)02d
command=php /var/www/posthider/artisan queue:work database --tries 1 --timeout 13000
autostart=true
autorestart=true
user=www-data
numprocs=3
redirect_stderr=true
stdout_logfile=/var/www/posthider/storage/logs/queue-workers.log

//Content ended
sudo supervisorctl reread
sudo supervisorctl update
sudo supervisorctl stop posthider-queue-workers:*
sudo supervisorctl start posthider-queue-workers:*

php artisan queue:restart
